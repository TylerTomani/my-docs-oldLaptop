from bs4 import BeautifulSoup
import requests

url = 'https://remoteok.com/'
req = requests.get(url, headers={'User-Agent': 'Mozilla/5.0'})
soup = BeautifulSoup(req.text, 'html.parser')
jobs = []
filtered_skills = [skill.strip() for skill in input('>').split(',')]

for tr in soup.find_all('tr', 'job'):
    if tr.find('td', 'source').a['href']:
        job = {}
        job['role'] = tr.find('td', 'company').a.h2.text.strip()
        job['company'] = tr.find('span', 'companyLink').h3.text.strip()
        job['tags'] = []
        for tag in tr.find_all('div', 'tag'):
            job['tags'].append(tag.text.replace('\n', '').strip())
        job['link'] = 'https://remoteok.com/' + tr.find('td', 'source').a['href']
        jobs.append(job)

for job in jobs:
    skill_found = False
    for tag in job['tags']:
        if any(skill.lower() in tag.lower() for skill in filtered_skills):
            skill_found = True
            break

    if not skill_found:
        print('Job without filtered skills:')
        print(job)
        print('\n')
