01

You can use open to read a html file and place it into a soup object, 

from bs4 import BeautifulSoup
with open('example-html.htm','r') as f:
content = f.read()
soup = BeautifulSoup(content, 'html.parser')

print(soup)

02

Get text with .text and attributes with ['<attrbName>'], like ['href'] square brackets

Use whatever methods to get the text you want, and, you can specify classes in the 2nd argument of soup.find('a', '<classname>') 

from bs4 import BeautifulSoup
with open('example-html.htm','r') as f:
content = f.read()
soup = BeautifulSoup(content, 'html.parser')

title = soup.title
print(title)
title = soup.title.text
print(title)
a_link1 = soup.find('a','sister ')
print(a_link1)
link = a_link1['href']
print(link)

03

.select

Allows you to use css selectors 

from bs4 import BeautifulSoup
with open('example-html.htm','r') as f:
content = f.read()
soup = BeautifulSoup(content, 'html.parser')
# print(soup)

story_as = soup.select('body > p > a.sister')
for a in story_as:
    print(a)
