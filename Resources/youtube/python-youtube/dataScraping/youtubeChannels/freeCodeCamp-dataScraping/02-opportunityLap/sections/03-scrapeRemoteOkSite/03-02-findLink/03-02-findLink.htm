01

find the link within the job container on https://remoteok.com/ <td class="source" → <a> tag, but notice the url in href is only the end of the url.

02

Add the url https://remoteok.com/ to the beggining of the url and create a key in the job {} for ['link']. Make sure to use [] bracket notation to get the the ['href'] attribute of the <a> tag. 

from bs4 import BeautifulSoup
import requests
url = 'https://remoteok.com/'
req = requests.get(url,headers={'User-Agent':'Moziila/5.0'})
soup = BeautifulSoup(req.text,'html.parser')
jobs=[]
for tr in soup.find_all('tr','job'):
    job = {}
    job['roll'] = tr.find('td','company').a.h2.text.strip()
    job['company'] = tr.find('span','companyLink').h3.text.strip()
    tags = []
    for tag in tr.find_all('div','tag'):
        tags.append(tag.text.replace('\n','').strip())
    job['tags'] = tags

    job['link'] = 'https://remoteok.com/' + tr.find('td','source').a['href']
    print(job['roll'])
    print(job['company'])
    print(job['tags'])
    print(job['link'])
    break
