from bs4 import BeautifulSoup
import requests
url = 'https://remoteok.com/'
req = requests.get(url, headers={'User-Agent': 'Mozilla/5.0'})
soup = BeautifulSoup(req.text,'html.parser')
jobs = []
for tr in soup.find_all('tr','job'):
    job = {}
    job['roll'] = tr.find('td','company').a.h2.text.strip()
    job['company'] = tr.find('span','companyLink').h3.text.strip()
    job['tags'] = []
    for tag in tr.find_all('div','tag'):
        job['tags'].append(tag.h3.text.replace('\n',''))
    print(job['roll'])
    print(job['company'])
    print(job['tags'])
    break
