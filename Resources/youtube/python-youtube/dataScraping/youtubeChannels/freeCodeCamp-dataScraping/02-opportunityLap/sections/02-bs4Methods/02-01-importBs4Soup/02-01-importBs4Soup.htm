from bs4 import BeautifulSoup

Create a soup object and insert the html by putting into a variable, and using 'html.parser' 

from bs4 import BeautifulSoup
html = ''' 
<html><head><title>The Doormouse's story</title></head>
<body>
<p class="story">Once upon a time there were three little sisters; and their names
were
<a href="https://example.com/elsie class="sister" id="link1">elsie</a>
<a href="https://example.com/lacie class="sister" id="link1">lacie</a> and
<a href="https://example.com/tillie class="sister" id="link1">tillie</a>
and theey lived in a wall </p>
</body>
</html>
'''
soup = BeautifulSoup(html, 'html.parser') 
print(soup)

We can now use the soup to query the html.
